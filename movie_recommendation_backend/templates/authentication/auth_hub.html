🔐 Authentication Flow Example

// 1. Register a new user const registerResponse = await fetch('/authentication/auth/register/', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: 'moviefan123', email: 'user@example.com', password: 'securepass123', first_name: 'John', last_name: 'Doe' }) }); // 2. Login to get tokens const loginResponse = await fetch('/authentication/auth/login/', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: 'moviefan123', password: 'securepass123' }) }); const { tokens } = await loginResponse.json(); // 3. Use token for authenticated requests const profileResponse = await fetch('/authentication/auth/users/', { headers: { 'Authorization': `Bearer ${tokens.access_token}` } });
